from kartograph import Kartograph

def myfilter(record):
    return record['name'] == "France"

cfg = {
  "proj": {
    "id": "mercator",
    "lon0": "auto"
    },
    "layers": {
        "mylayer": {
            "src": "subunits.shp",
            "filter": myfilter
        }
    }
}
    
K = Kartograph()
K.generate(cfg, outfile='mymap.svg')
